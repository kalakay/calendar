 \def\printcalendar#1#2{%
   \luadirect{
      local calendar =require("i18n.calendar")
      calendar.makeCalendar(#1,#2)  -- month/year
      }%   
 }
 
 % #1 yesr
 \def\printfullcalendar#1{%
   \luadirect{
      local calendar =require("i18n.calendar")
      calendar.yearCalendar(#1)  -- month/year
      }%   
 }
